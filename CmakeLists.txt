cmake_minimum_required(VERSION 3.13)

include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)
include($ENV{PICO_EXTRAS_PATH}/external/pico_extras_import.cmake)

project(rp2040_dac C CXX ASM)
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)
set(PICO_TINYUSB_PATH ${CMAKE_CURRENT_LIST_DIR}/tinyusb)
set(tinyusb_config_dir "D:/FILE/pico/dac/usb")

message(STATUS "Using upstream TinyUSB at: ${PICO_TINYUSB_PATH}")
message(STATUS "SDK EXTRA at: ${PICO_EXTRAS_PATH}")
# Set TinyUSB config file with absolute path before pico_sdk_init

pico_sdk_init()

add_executable(rp2040_dac
    common/i2s.c
    usb/usb_descriptors.c
    usb/uac.c
    pico-ssd1306/ssd1306.c
    BME280_SensorAPI/bme280.c
    main.c
)

    BME280_SensorAPI
